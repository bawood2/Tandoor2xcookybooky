\begin{recipe}[preparationtime = {\unit[30]{min}}, bakingtime={\unit[15]{min}}, portion = \portion{4\hspace{0.2em}}, source = \href{Schnelles für Mutter und Kind  Seite 54}{Schnelles für Mutter und Kind  Seite 54}]{Asiapfanne mit Reis}
\graph{big=Pictures/Asiapfanne mit Reis.jpg}

\ingredients{%
\unit[400]{g} & Rinderfilet {\textit{\footnotesize in Streifen}} \\
\unit[700]{g} & Spitzkohl {\textit{\footnotesize in Streifen}} \\
\unit[2] & rote Paprikas {\textit{\footnotesize gewürfelt}} \\
\unit[2] & gelbe Zwiebeln {\textit{\footnotesize gewürfelt}} \\
\unit[2]{Esslöffel} & Sesam\\
\unit[4]{Esslöffel} & Sesamöl\\
\unit[2]{Prisen} & Salz\\
\unit[4]{Esslöffel} & Ketchup\\
\unit[2]{Esslöffel} & Sojasauce\\
\unit[2]{-} & Chilischote\\
\unit[6]{Stängel} & Koriander\\
\unit[2]{Esslöffel} & Sherry\\
\unit[1\nicefrac{1}{2}]{Tassen} & Reis\\
}


\preparation{%
\step{Fleisch in dünne Streifen schneiden. Kohl und Paprikaschote waschen und zerkleinern. Kohl in feine Streifen und Paprika in kleine Stücke schneiden.}
\step{Sesam im Wok ohne Öl kurz anrösten und rausnehmen. Im gleichen Wok Öl erhitzen und das Fleisch rundum kräftig anbraten, salzen und herausnehmen.}
\step{Paprika, Kohl und Zwiebeln in die Pfanne geben und darin anbraten. Bei mittlerer Hitze zugedeckt ca. 5~Min. schmoren.}
\step{Ketchup, Sojasauce und pro Portion 3 EL Wasser verrühren und zum Gemüse geben. Fleisch ebenfalls zugeben und alles aufkochen. Chili und Koriander klein hacken.}
\step{Kinderportion zur Seite stellen und mit etwas Sesam bestreuen. Zum Rest Sherry und Chili zum Gemüse geben und aufkochen. Mit Koriander und Sesam garnieren und Servieren.}
}

\hint{Wok}
\end{recipe}
